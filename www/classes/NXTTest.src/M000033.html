        def test_motors_multiple
                @nxt.motors_abc do |m|
                        m.reset_tacho
                end
        
                @nxt.motors_abc do |m|
                        m.forward(:degrees => 360, :power => 10)
                end
                
                @nxt.motors_abc do |m|
                        assert_in_delta(360, m.read_state[:degree_count], 20)
                end
