  def test_run_by_seconds
        @motors.each do |m|
               m.reset_tacho
               m.forward(:time => 3, :power => 15)
               state = m.read_state
               assert_in_delta(344, state[:degree_count], 50)
               
               m.reset_tacho
               m.backward(:time => 3, :power => 15)
               state = m.read_state
               assert_in_delta(-344, state[:degree_count], 50)
          end
  end
